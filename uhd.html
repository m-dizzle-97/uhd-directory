        <h1>Google Sheet → Searchable HTML table</h1>

      </div>

    </header>


    <div class="toolbar">

      <input id="searchInput" class="search" type="text" placeholder="Search (type to filter across all columns)" />

      <select id="pageSizeSelect">

        <option value="10">10 rows</option>

        <option value="25">25 rows</option>

        <option value="50">50 rows</option>

        <option value="100">100 rows</option>

      </select>

      <table id="dataTable" aria-live="polite"><thead></thead><tbody></tbody></table>

    </div>


    <div class="footer">

      <p>If nothing shows up — ensure the sheet is published or shared publicly (Anyone with link can view).</p>

    </div>

  <script>

    // ---------- Configuration & helpers ----------

    const AUTO_LOAD_URL = 'https://docs.google.com/spreadsheets/d/1mvFObcnflcZOeknieOtQdgCV1l_LZlljtQS1Vdx2Xgs/export?format=csv&gid=1911649658';


    const statusEl = document.createElement('div');

    document.querySelector('.controls')?.appendChild(statusEl);

    const searchInput = document.getElementById('searchInput');

    const dataTable = document.getElementById('dataTable');

    const pageSizeSelect = document.getElementById('pageSizeSelect');


    let rawRows = [];

    let headers = [];

    let filteredRows = [];

    let sortState = {col: null, dir: 1};


    function setStatus(t){ statusEl.textContent = t }


    function parseCSV(text){

      const rows = [];

      let cur = '';

      let row = [];

      let inQuotes = false;

      for(let i=0;i<text.length;i++){

        const ch = text[i];

        const nxt = text[i+1];

        if(ch === '"'){

          if(inQuotes && nxt === '"'){ cur += '"'; i++; continue; }

          inQuotes = !inQuotes;

          continue;

        }

        if(ch === ',' && !inQuotes){ row.push(cur); cur=''; continue; }

        if((ch === '\\n' || ch === '\\r') && !inQuotes){

          if(cur !== '' || row.length>0){ row.push(cur); cur=''; rows.push(row); row = []; }

          if(ch === '\\r' && nxt === '\\n') i++;

          continue;

        }

        cur += ch;

      }

      if(cur !== '' || row.length>0){ row.push(cur); rows.push(row); }

      return rows;

    }


    function loadFromUrl(url){

      setStatus('Fetching data …');

      fetch(url).then(r=>{

        if(!r.ok) throw new Error('Network response not ok: '+r.status);

        return r.text();

      }).then(text=>{

        const rows = parseCSV(text);

        if(rows.length === 0) throw new Error('No rows found in the CSV');

        headers = rows[0].map(h=>h.trim());

        rawRows = rows.slice(1).map(r=>{

          const obj = {};

          for(let i=0;i<headers.length;i++) obj[headers[i]||`col${i+1}`] = r[i] ?? '';

          return obj;

        });

        filteredRows = rawRows.slice();

        renderTable();

        setStatus('Loaded ' + rawRows.length + ' rows.');

      }).catch(err=>{

        console.error(err);

        setStatus('Error: ' + err.message + ' — check sheet sharing / publish settings.');

      });

    }


    function renderTable(){

      const thead = dataTable.querySelector('thead');

      const tbody = dataTable.querySelector('tbody');

      thead.innerHTML = '';

      tbody.innerHTML = '';


      const tr = document.createElement('tr');

      headers.forEach((h,ci)=>{

        const th = document.createElement('th');

        th.textContent = h || `Column ${ci+1}`;

        th.classList.add('sortable');

        th.addEventListener('click', ()=>{

          if(sortState.col === ci) sortState.dir *= -1; else { sortState.col = ci; sortState.dir = 1; }

          sortByColumn(ci, sortState.dir);

          renderTable();

        });

        tr.appendChild(th);

      });

      thead.appendChild(tr);


      const pageSize = parseInt(pageSizeSelect.value,10);

      const q = (searchInput.value||'').toLowerCase().trim();

      filteredRows = rawRows.filter(r=>{

        if(!q) return true;

        return headers.some(h=> (r[h]||'').toString().toLowerCase().includes(q));

      });


      const rowsToShow = filteredRows.slice(0, pageSize);

      rowsToShow.forEach(rowObj=>{

        const r = document.createElement('tr');

        headers.forEach(h=>{

          const td = document.createElement('td');

          td.textContent = rowObj[h] ?? '';

          r.appendChild(td);

        });

        tbody.appendChild(r);

      });


      if(filteredRows.length === 0){

        const r = document.createElement('tr');

        const td = document.createElement('td');

        td.colSpan = Math.max(1, headers.length);

        td.textContent = 'No rows match your filter.';

        r.appendChild(td);

        tbody.appendChild(r);

      }

    }


    function sortByColumn(ci, dir){

      const key = headers[ci];

      rawRows.sort((a,b)=>{

        const A = (a[key]||'').toString();

        const B = (b[key]||'').toString();

        const An = parseFloat(A.replace(/[^0-9.-]/g,''));

        const Bn = parseFloat(B.replace(/[^0-9.-]/g,''));

        if(!isNaN(An) && !isNaN(Bn)) return (An - Bn) * dir;

        return A.localeCompare(B) * dir;

      });

    }


    searchInput.addEventListener('input', renderTable);

    pageSizeSelect.addEventListener('change', renderTable);


    // Auto-load your sheet

    loadFromUrl(AUTO_LOAD_URL);

  </script>
